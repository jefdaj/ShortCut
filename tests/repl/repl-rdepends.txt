Welcome to the ShortCut interpreter!
Type :help for a list of the available commands.
shortcut >> v1 = "one"
shortcut >> v2 = "two"
shortcut >> v3 = [v1, v2]
shortcut >> :show
v1 = "one"
v2 = "two"
v3 = [v1, v2]
shortcut >> :rdepends v1
v3
shortcut >> :rdepends v2
v3
shortcut >> :rdepends v3

shortcut >> v4 = v3
shortcut >> :rdepends v3
v4
shortcut >> v5 = v4
shortcut >> :rdepends v3
v4, v5
shortcut >> :q
