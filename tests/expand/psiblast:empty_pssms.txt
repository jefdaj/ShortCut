query = load_faa_path (to_path "examples/sequences/Mycoplasma_genitalium_single.faa")
maga = load_faa_path (to_path "examples/sequences/Mycoplasma_agalactiae_small.faa")
magadb = makeblastdb_faa maga
mgen = load_faa_path (to_path "examples/sequences/Mycoplasma_genitalium_protein_refseq.faa")
mgendb = makeblastdb_faa mgen
cutoff = 0.5
pssmmaga = psiblast_train_db cutoff query magadb
pssmmgen = psiblast_train_db cutoff query mgendb
cutoffs = [1, 0.5, 0.1, 1.0e-2, 1.0e-3, 1.0e-5, 1.0e-10, 1.0e-20, 1.0e-50, 1.0e-100]
pssmsmaga = replace_each pssmmaga cutoff cutoffs
pssmsmgen = replace_each pssmmgen cutoff cutoffs
hitsmaga = psiblast_each_pssm cutoff pssmsmaga maga
hitsmgen = psiblast_each_pssm cutoff pssmsmgen mgen
result = hitsmgen
