mgen = load_fna (gbk_to_fna_rawids "cds"
                                   (load_gbk "examples/sequences/Mycoplasma_genitalium_M2321.gbk"))
mbov = load_fna (gbk_to_fna_rawids "cds"
                                   (load_gbk "examples/sequences/Mycoplasma_bovis_HB0801-P115.gbk"))
single = extract_queries (blastn 1.0e-5 mbov mgen)
