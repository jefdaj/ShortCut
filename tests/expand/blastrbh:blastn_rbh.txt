mgen = load_fna "examples/sequences/Mycoplasma_genitalium_M2321_5genes.fna"
maga = load_fna "examples/sequences/Mycoplasma_genitalium_M2321_5genes.fna"
n = 1.0e-5
fwd = blastn_db n mgen (makeblastdb_nucl_all [maga])
rev1 = blastn_db n maga (makeblastdb_nucl_all [mgen])
rev2 = blastn_db n maga (makeblastdb_nucl mgen)
rbh1 = reciprocal_best fwd rev1
rbh2 = reciprocal_best fwd rev2
rbh3 = reciprocal_best (blastn_db n mgen (makeblastdb_nucl_all [maga]))
                       (blastn_db n maga (makeblastdb_nucl mgen))
rbh4 = reciprocal_best (blastn_db n maga (makeblastdb_nucl_all [mgen]))
                       (blastn_db n mgen (makeblastdb_nucl maga))
lists = extract_queries_each [rbh1, rbh2, rbh3] | (extract_targets_each [rbh4])
result = any lists ~ (all lists)
