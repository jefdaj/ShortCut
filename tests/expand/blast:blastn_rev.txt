maga5 = load_fna "examples/sequences/Mycoplasma_genitalium_M2321_5genes.fna"
maga = load_fna "examples/sequences/Mycoplasma_genitalium_M2321_5genes.fna"
n = 1.0e-5
singleFwd = extract_queries (blastn_db n maga5 (makeblastdb_nucl_all [maga]))
mappedFwd = extract_queries_each (blastn_db_each n
                                                 maga5
                                                 (makeblastdb_nucl_all_each (singletons [maga])))
singleRev = extract_queries (blastn_db n maga5 (makeblastdb_nucl_all [maga]))
mappedRev = extract_queries_each (blastn_rev_each n maga [maga5])
single = singleFwd | singleRev
mapped = mappedFwd | mappedRev
result = [single] | mapped
