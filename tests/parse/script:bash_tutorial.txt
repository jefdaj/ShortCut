Script
    { sAssigns =
        [ Assign
            { aVar = Var ( RepID Nothing ) "s"
            , aExpr = Lit str "a string"
            }
        , Assign
            { aVar = Var ( RepID Nothing ) "n"
            , aExpr = Lit num "1.4e7"
            }
        , Assign
            { aVar = Var ( RepID Nothing ) "nl"
            , aExpr = Lst num Nothing []
                [ Lit num "1.0"
                , Lit num "2.0"
                , Lit num "3.0"
                , Lit num "100.0"
                ]
            }
        , Assign
            { aVar = Var ( RepID Nothing ) "sl"
            , aExpr = Lst str Nothing []
                [ Lit str "another string"
                , Lit str "and another"
                , Lit str "and a third"
                ]
            }
        , Assign
            { aVar = Var ( RepID Nothing ) "nll"
            , aExpr = Lst num.list Nothing
                [ Var ( RepID Nothing ) "nl" ]
                [ Ref num.list Nothing []
                    ( Var ( RepID Nothing ) "nl" )
                , Lst num Nothing []
                    [ Lit num "1.0"
                    , Lit num "2.0"
                    , Lit num "3.0"
                    ]
                ]
            }
        , Assign
            { aVar = Var ( RepID Nothing ) "txt"
            , aExpr = Fun untyped Nothing
                [ Var ( RepID Nothing ) "s"
                , Var ( RepID Nothing ) "n"
                , Var ( RepID Nothing ) "nl"
                , Var ( RepID Nothing ) "sl"
                , Var ( RepID Nothing ) "nll"
                ] "run_script"
                [ Lit str "examples/scripts/custom-bash-script.sh"
                , Lst untyped Nothing
                    [ Var ( RepID Nothing ) "s"
                    , Var ( RepID Nothing ) "n"
                    , Var ( RepID Nothing ) "nl"
                    , Var ( RepID Nothing ) "sl"
                    , Var ( RepID Nothing ) "nll"
                    ]
                    [ Ref str Nothing []
                        ( Var ( RepID Nothing ) "s" )
                    , Ref num Nothing []
                        ( Var ( RepID Nothing ) "n" )
                    , Ref num.list Nothing []
                        ( Var ( RepID Nothing ) "nl" )
                    , Ref str.list Nothing []
                        ( Var ( RepID Nothing ) "sl" )
                    , Ref num.list.list Nothing
                        [ Var ( RepID Nothing ) "nl" ]
                        ( Var ( RepID Nothing ) "nll" )
                    ]
                ]
            }
        , Assign
            { aVar = Var ( RepID Nothing ) "result"
            , aExpr = Ref untyped Nothing
                [ Var ( RepID Nothing ) "s"
                , Var ( RepID Nothing ) "n"
                , Var ( RepID Nothing ) "nl"
                , Var ( RepID Nothing ) "sl"
                , Var ( RepID Nothing ) "nll"
                ]
                ( Var ( RepID Nothing ) "txt" )
            }
        ]
    , sResult = Just
        ( Ref untyped Nothing
            [ Var ( RepID Nothing ) "s"
            , Var ( RepID Nothing ) "n"
            , Var ( RepID Nothing ) "nl"
            , Var ( RepID Nothing ) "sl"
            , Var ( RepID Nothing ) "nll"
            ]
            ( Var ( RepID Nothing ) "txt" )
        )
    }
